1//! # Minimal Unsafe C API for OpenMP Parser
2//!
3//! This module provides a direct pointer-based C FFI with minimal unsafe code.
4//!
5//! ## Design Philosophy: Direct Pointers vs Handles
6//!
7//! We use **raw C pointers** to Rust structs instead of opaque handles for:
8//! - **Idiomatic C API**: Standard malloc/free pattern familiar to C programmers
9//! - **Simple memory model**: No global registry, no handle bookkeeping
10//! - **Easy integration**: Works naturally with C/C++ code
11//! - **Minimal code**: 632 lines vs 4000+ lines of handle management
12//!
13//! ## Safety Analysis: 18 Unsafe Blocks (~60 lines)
14//!
15//! All unsafe blocks are:
16//! - **NULL-checked** before dereferencing
17//! - **Documented** with explicit safety requirements
18//! - **Isolated** only at FFI boundary, never in business logic
19//! - **Auditable**: ~0.9% of file (60 unsafe lines / 632 total)
20//!
21//! ## Learning Rust: Why Unsafe is Needed at FFI Boundary
22//!
23//! 1. **C strings** → Rust strings: `CStr::from_ptr()` requires unsafe
24//! 2. **Memory ownership** transfer: `Box::into_raw()` / `Box::from_raw()`
25//! 3. **Raw pointer dereferencing**: C passes pointers, we must dereference
26//!
27//! Safe Rust cannot verify C's guarantees, so we explicitly document them.
28//!
29//! ## C Caller Responsibilities
30//!
31//! C callers MUST:
32//! - ✅ Check for NULL returns before use
33//! - ✅ Call `_free()` functions to prevent memory leaks
34//! - ✅ Never use pointers after calling `_free()`
35//! - ✅ Pass only valid null-terminated strings
36//! - ✅ Not modify strings returned by this API
37

80/// Clause-specific data stored in a C union
81///
82/// Learning Rust: Why ManuallyDrop?
83/// =================================
84/// Unions in Rust don't know which variant is active, so they can't
85/// automatically call destructors. ManuallyDrop prevents automatic drops
86/// and lets us manually free resources when we know which variant is active.
87#[repr(C)]
88union ClauseData {
89    schedule: ManuallyDrop<ScheduleData>,
90    reduction: ManuallyDrop<ReductionData>,
91    default: i32,
92    variables: *mut OmpStringList,
93}

131/// Parse an OpenMP directive from a C string.
132///
133/// ## Parameters
134/// - `input`: Null-terminated C string containing the directive
135///
136/// ## Returns
137/// - Pointer to `OmpDirective` on success
138/// - NULL on parse failure or NULL input
139///
140/// ## Safety
141/// Caller must:
142/// - Pass valid null-terminated C string or NULL
143/// - Call `roup_directive_free()` on the returned pointer
144///
145/// ## Example
146/// ```c
147/// OmpDirective* dir = roup_parse("#pragma omp parallel");
148/// if (dir) {
149///     // use directive
150///     roup_directive_free(dir);
151/// }
152/// ```
153#[no_mangle]
154pub extern "C" fn roup_parse(input: *const c_char) -> *mut OmpDirective {
155    // NULL check
156    if input.is_null() {
157        return ptr::null_mut();
158    }
159
160    // UNSAFE BLOCK 1: Convert C string to Rust &str
161    // Safety: Caller guarantees valid null-terminated C string
162    let c_str = unsafe { CStr::from_ptr(input) };
163
164    let rust_str = match c_str.to_str() {
165        Ok(s) => s,
166        Err(_) => return ptr::null_mut(), // Invalid UTF-8
167    };
168
169    // Parse using safe Rust parser
170    let directive = match parse_omp_directive(rust_str) {
171        Ok((_, dir)) => dir,
172        Err(_) => return ptr::null_mut(), // Parse error
173    };
174
175    // Convert to C-compatible format
176    let c_directive = OmpDirective {
177        name: allocate_c_string(directive.name),
178        clauses: directive
179            .clauses
180            .into_iter()
181            .map(|c| convert_clause(&c))
182            .collect(),
183    };
184
185    // UNSAFE BLOCK 2: Convert Box to raw pointer for C
186    // Safety: Caller will call roup_directive_free() to deallocate
187    Box::into_raw(Box::new(c_directive))
188}
189
190/// Free a directive allocated by `roup_parse()`.
191///
192/// ## Safety
193/// - Must only be called once per directive
194/// - Pointer must be from `roup_parse()`
195/// - Do not use pointer after calling this function
196#[no_mangle]
197pub extern "C" fn roup_directive_free(directive: *mut OmpDirective) {
198    if directive.is_null() {
199        return;
200    }
201
202    // UNSAFE BLOCK 3: Convert raw pointer back to Box for deallocation
203    // Safety: Pointer came from Box::into_raw in roup_parse
204    unsafe {
205        let boxed = Box::from_raw(directive);
206
207        // Free the name string (was allocated with CString::into_raw)
208        if !boxed.name.is_null() {
209            drop(CString::from_raw(boxed.name as *mut c_char));
210        }
211
212        // Free clause data
213        for clause in &boxed.clauses {
214            free_clause_data(clause);
215        }
216
217        // Box is dropped here, freeing memory
218    }
219}

221/// Free a clause.
222#[no_mangle]
223pub extern "C" fn roup_clause_free(clause: *mut OmpClause) {
224    if clause.is_null() {
225        return;
226    }
227
228    unsafe {
229        let boxed = Box::from_raw(clause);
230        free_clause_data(&boxed);
231    }
232}

296/// Get next clause from iterator.
297///
298/// ## Parameters
299/// - `iter`: Iterator from `roup_directive_clauses_iter()`
300/// - `out`: Output pointer to write clause pointer
301///
302/// ## Returns
303/// - 1 if clause available (written to `out`)
304/// - 0 if no more clauses or NULL inputs
305#[no_mangle]
306pub extern "C" fn roup_clause_iterator_next(
307    iter: *mut OmpClauseIterator,
308    out: *mut *const OmpClause,
309) -> i32 {
310    // NULL checks
311    if iter.is_null() || out.is_null() {
312        return 0;
313    }
314
315    // UNSAFE BLOCK 5: Dereference iterator
316    // Safety: Caller guarantees valid iterator pointer
317    unsafe {
318        let iterator = &mut *iter;
319
320        if iterator.index >= iterator.clauses.len() {
321            return 0; // No more items
322        }
323
324        let clause_ptr = iterator.clauses[iterator.index];
325        iterator.index += 1;
326
327        // UNSAFE BLOCK 6: Write to output pointer
328        // Safety: Caller guarantees valid output pointer
329        *out = clause_ptr;
330        1
331    }
332}

339    }

341    unsafe {
342        drop(Box::from_raw(iter));
343    }

524/// Allocate a C string from Rust &str.
525///
526/// Creates a heap-allocated, null-terminated C string from Rust string.
527/// The returned pointer must be freed with CString::from_raw() later.
528///
529/// ## How it works:
530/// 1. CString::new() creates null-terminated copy
531/// 2. into_raw() gives us ownership of the pointer
532/// 3. Caller (or roup_directive_free) must eventually free it
533fn allocate_c_string(s: &str) -> *const c_char {
534    let c_string = std::ffi::CString::new(s).unwrap();
535    c_string.into_raw() as *const c_char
536}

538/// Convert Rust Clause to C-compatible OmpClause.
539///
540/// Maps clause names to integer kind codes (C doesn't have Rust enums).
541/// Each clause type gets a unique ID and appropriate data representation.
542///
543/// ## Clause Kind Mapping:
544/// - 0 = num_threads    - 6 = reduction
545/// - 1 = if             - 7 = schedule
546/// - 2 = private        - 8 = collapse
547/// - 3 = shared         - 9 = ordered
548/// - 4 = firstprivate   - 10 = nowait
549/// - 5 = lastprivate    - 11 = default
550/// - 999 = unknown
551fn convert_clause(clause: &Clause) -> OmpClause {
552    let (kind, data) = match clause.name {
553        "num_threads" => (0, ClauseData { default: 0 }),
554        "if" => (1, ClauseData { default: 0 }),
555        "private" => (
556            2,
557            ClauseData {
558                variables: ptr::null_mut(),
559            },
560        ),
561        "shared" => (
562            3,
563            ClauseData {
564                variables: ptr::null_mut(),
565            },
566        ),
567        "firstprivate" => (
568            4,
569            ClauseData {
570                variables: ptr::null_mut(),
571            },
572        ),
573        "lastprivate" => (
574            5,
575            ClauseData {
576                variables: ptr::null_mut(),
577            },
578        ),
579        "reduction" => {
580            let operator = parse_reduction_operator(clause);
581            (
582                6,
583                ClauseData {
584                    reduction: ManuallyDrop::new(ReductionData { operator }),
585                },
586            )
587        }
588        "schedule" => {
589            let schedule_kind = parse_schedule_kind(clause);
590            (
591                7,
592                ClauseData {
593                    schedule: ManuallyDrop::new(ScheduleData {
594                        kind: schedule_kind,
595                    }),
596                },
597            )
598        }
599        "collapse" => (8, ClauseData { default: 0 }),
600        "ordered" => (9, ClauseData { default: 0 }),
601        "nowait" => (10, ClauseData { default: 0 }),
602        "default" => {
603            let default_kind = parse_default_kind(clause);
604            (
605                11,
606                ClauseData {
607                    default: default_kind,
608                },
609            )
610        }
611        _ => (999, ClauseData { default: 0 }), // Unknown
612    };
613
614    OmpClause { kind, data }
615}

617/// Parse reduction operator from clause arguments.
618///
619/// Extracts the operator from reduction clause like "reduction(+: sum)".
620/// Returns integer code for the operator type.
621///
622/// ## Operator Codes:
623/// - 0 = +  (addition)      - 5 = ^  (bitwise XOR)
624/// - 1 = -  (subtraction)   - 6 = && (logical AND)
625/// - 2 = *  (multiplication) - 7 = || (logical OR)
626/// - 3 = &  (bitwise AND)   - 8 = min
627/// - 4 = |  (bitwise OR)    - 9 = max
628fn parse_reduction_operator(clause: &Clause) -> i32 {
629    // Look for operator in clause kind
630    if let ClauseKind::Parenthesized(args) = clause.kind {
631        if args.contains('+') && !args.contains("++") {
632            return 0; // Plus
633        } else if args.contains('-') && !args.contains("--") {
634            return 1; // Minus
635        } else if args.contains('*') {
636            return 2; // Times
637        } else if args.contains('&') && !args.contains("&&") {
638            return 3; // BitwiseAnd
639        } else if args.contains('|') && !args.contains("||") {
640            return 4; // BitwiseOr
641        } else if args.contains('^') {
642            return 5; // BitwiseXor
643        } else if args.contains("&&") {
644            return 6; // LogicalAnd
645        } else if args.contains("||") {
646            return 7; // LogicalOr
647        } else if args.contains("min") {
648            return 8; // Min
649        } else if args.contains("max") {
650            return 9; // Max
651        }
652    }
653    0 // Default to plus
654}

656/// Parse schedule kind from clause arguments.
657///
658/// Extracts schedule type from clause like "schedule(dynamic, 4)".
659/// Returns integer code for the schedule policy.
660///
661/// ## Schedule Codes:
662/// - 0 = static   (default, divide iterations evenly)
663/// - 1 = dynamic  (distribute at runtime)
664/// - 2 = guided   (decreasing chunk sizes)
665/// - 3 = auto     (compiler decides)
666/// - 4 = runtime  (OMP_SCHEDULE environment variable)
667fn parse_schedule_kind(clause: &Clause) -> i32 {
668    if let ClauseKind::Parenthesized(args) = clause.kind {
669        if args.contains("static") {
670            return 0;
671        } else if args.contains("dynamic") {
672            return 1;
673        } else if args.contains("guided") {
674            return 2;
675        } else if args.contains("auto") {
676            return 3;
677        } else if args.contains("runtime") {
678            return 4;
679        }
680    }
681    0 // Default to static
682}

684/// Parse default clause data-sharing attribute.
685///
686/// Extracts the default sharing from clause like "default(shared)".
687/// Returns integer code for the default policy.
688///
689/// ## Default Codes:
690/// - 0 = shared (all variables shared by default)
691/// - 1 = none   (must explicitly declare all variables)
692fn parse_default_kind(clause: &Clause) -> i32 {
693    if let ClauseKind::Parenthesized(args) = clause.kind {
694        if args.contains("shared") {
695            return 0;
696        } else if args.contains("none") {
697            return 1;
698        }
699    }
700    0 // Default to shared
701}

703/// Convert directive name to kind enum code.
704///
705/// Maps directive names (parallel, for, task, etc.) to integer codes
706/// so C code can use switch statements instead of string comparisons.
707///
708/// ## Directive Codes:
709/// - 0 = parallel     - 5 = critical
710/// - 1 = for          - 6 = atomic
711/// - 2 = sections     - 7 = barrier
712/// - 3 = single       - 8 = master
713/// - 4 = task         - 9 = teams
714/// - 10 = target      - 11 = distribute
715/// - -1 = NULL/unknown
716fn directive_name_to_kind(name: *const c_char) -> i32 {
717    if name.is_null() {
718        return -1;
719    }
720
721    // UNSAFE BLOCK 11: Read directive name
722    // Safety: name pointer is valid (from our own allocation)
723    unsafe {
724        let c_str = CStr::from_ptr(name);
725        let name_str = c_str.to_str().unwrap_or("");
726
727        match name_str {
728            "parallel" => 0,
729            "parallel for" => 0, // Composite: treat as parallel
730            "parallel sections" => 0,
731            "for" => 1,
732            "sections" => 2,
733            "single" => 3,
734            "task" => 4,
735            "master" => 5,
736            "critical" => 6,
737            "barrier" => 7,
738            "taskwait" => 8,
739            "taskgroup" => 9,
740            "atomic" => 10,
741            "flush" => 11,
742            "ordered" => 12,
743            "target" => 13,
744            "target teams" => 13, // Composite: treat as target
745            "teams" => 14,
746            "teams distribute" => 14, // Composite: treat as teams
747            "distribute" => 15,
748            "metadirective" => 16,
749            _ => 999, // Unknown
750        }
751    }
752}

754/// Free clause-specific data.
755/// Free clause-specific data (internal helper).
756///
757/// Handles cleanup for union types where different variants need different
758/// cleanup strategies. Currently only variable lists need explicit freeing.
759///
760/// ## Design Note:
761/// Most clause data (schedule, reduction, default) are Copy types that don't
762/// need cleanup. Only heap-allocated variable lists need explicit freeing.
763///
764/// ## IMPORTANT: ClauseData is a C union!
765/// The ClauseData union has 4 fields, but only ONE is active per clause:
766///   - `variables: *mut OmpStringList` - used by kinds 2-5 (private/shared/firstprivate/lastprivate)
767///   - `reduction: ReductionData` - used by kind 6 (reduction operator only, NO variables pointer)
768///   - `schedule: ScheduleData` - used by kind 7 (schedule policy only)
769///   - `default: i32` - used by other kinds
770///
771/// Reduction clauses (kind 6) do NOT use the `variables` field. Trying to free
772/// clause.data.variables on a reduction clause would read garbage memory from the
773/// wrong union variant (the bytes of ReductionData::operator interpreted as a pointer).
774fn free_clause_data(clause: &OmpClause) {
775    unsafe {
776        // Free variable lists if present
777        // Clause kinds with variable lists (see convert_clause):
778        //   2 = private, 3 = shared, 4 = firstprivate, 5 = lastprivate
779        // Other kinds use different union fields:
780        //   6 = reduction (uses .reduction field, NOT .variables)
781        //   7 = schedule (uses .schedule field, NOT .variables)
782        if clause.kind >= 2 && clause.kind <= 5 {
783            let vars_ptr = clause.data.variables;
784            if !vars_ptr.is_null() {
785                roup_string_list_free(vars_ptr);
786            }
787        }
788    }
789}
