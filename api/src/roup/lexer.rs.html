1/// Lexer module - tokenizes OpenMP pragma directives
2///
3/// Learning Rust: Parser Combinators with nom
4/// ===========================================
5/// nom is a parser combinator library
6/// - Parsers are functions that consume input and return results
7/// - Combine small parsers to build complex ones
8/// - Type: IResult<Input, Output, Error>
9
10/// Learning Rust: Type Aliases
11/// ============================
12/// IResult is nom's result type for parser functions
13/// IResult<&str, &str> means:
14/// - Input: &str (string slice to parse)
15/// - Output: &str (parsed token)
16/// - Returns: Ok((remaining_input, parsed_output)) or Err
17use nom::IResult;

22/// nom::bytes::complete::take_while1 - matches while predicate is true

25/// Language format for parsing
26#[derive(Debug, Clone, Copy, PartialEq, Eq)]
27pub enum Language {
28    /// C/C++ language with #pragma omp
29    C,
30    /// Fortran free-form with !$OMP sentinel
31    FortranFree,
32    /// Fortran fixed-form with !$OMP or C$OMP in columns 1-6
33    FortranFixed,
34}
35
36impl Default for Language {
37    fn default() -> Self {
38        Language::C
39    }
40}

42/// Check if a character is valid in an identifier
43///
44/// Learning Rust: Closures and Function Pointers
45/// ==============================================
46/// This function can be used as a predicate
47/// take_while1 accepts: fn(char) -> bool
48pub fn is_identifier_char(c: char) -> bool {
49    c.is_alphanumeric() || c == '_'
50}

57/// Parse "#pragma" keyword

73/// Parse Fortran free-form sentinel "!$OMP" (case-insensitive)
74///
75/// Supports leading whitespace before the sentinel (common for indented code):
76/// - "!$OMP PARALLEL" -> matches
77/// - "    !$OMP PARALLEL" -> matches (leading spaces consumed)
78/// - "  \t!$OMP DO" -> matches (mixed whitespace consumed)
79pub fn lex_fortran_free_sentinel(input: &str) -> IResult<&str, &str> {
80    // Skip optional leading whitespace (common in indented Fortran code)
81    let (after_space, _) = skip_space_and_comments(input)?;
82
83    // Optimize: check only first 5 characters instead of entire input
84    let matches = after_space
85        .get(..5)
86        .map_or(false, |s| s.eq_ignore_ascii_case("!$omp"));
87
88    if matches {
89        Ok((&after_space[5..], &after_space[..5]))
90    } else {
91        Err(nom::Err::Error(nom::error::Error::new(
92            input,
93            nom::error::ErrorKind::Tag,
94        )))
95    }
96}

98/// Parse Fortran fixed-form sentinel "!$OMP" or "C$OMP" in columns 1-6 (case-insensitive)
99///
100/// Supports leading whitespace before the sentinel:
101/// - "!$OMP PARALLEL" -> matches
102/// - "    !$OMP PARALLEL" -> matches (leading spaces consumed)
103/// - "C$OMP DO" -> matches
104/// - "*$OMP END PARALLEL" -> matches
105pub fn lex_fortran_fixed_sentinel(input: &str) -> IResult<&str, &str> {
106    // Skip optional leading whitespace (common in indented Fortran code)
107    let (after_space, _) = skip_space_and_comments(input)?;
108
109    // Optimize: check only first 5 characters instead of entire input
110    let first_5 = after_space.get(..5);
111    let matches = first_5.map_or(false, |s| {
112        s.eq_ignore_ascii_case("!$omp")
113            || s.eq_ignore_ascii_case("c$omp")
114            || s.eq_ignore_ascii_case("*$omp")
115    });
116
117    if matches {
118        Ok((&after_space[5..], &after_space[..5]))
119    } else {
120        Err(nom::Err::Error(nom::error::Error::new(
121            input,
122            nom::error::ErrorKind::Tag,
123        )))
124    }
125}

127// ============================================================================
128// Fortran Continuation Lines - NOT SUPPORTED BY DESIGN
129// ============================================================================
130//
131// ROUP does NOT support multi-line Fortran directives with continuation (&).
132//
133// Example of UNSUPPORTED input:
134//     !$OMP PARALLEL DO &
135//     !$OMP   PRIVATE(I,J)
136//
137// Design Decision:
138// - Continuation handling requires stateful multi-line parsing
139// - Users must provide complete single-line directives
140// - Applies to BOTH C and Fortran (no multi-line #pragma either)
141//
142// Workaround:
143// - Preprocess multi-line directives into single lines before calling parse()
144// - Example: "!$OMP PARALLEL DO PRIVATE(I,J)" (all on one line)
145//
146// See: src/parser/mod.rs Parser::parse() for rationale
147// ============================================================================

149/// Parse an identifier (directive or clause name)
150///
151/// Learning Rust: Higher-Order Functions
152/// ======================================
153/// take_while1 is a higher-order function
154/// It takes a function (predicate) and returns a parser
155/// The parser consumes chars while predicate is true
156fn lex_identifier(input: &str) -> IResult<&str, &str> {
157    take_while1(is_identifier_char)(input)
158}
159
160/// Parse a directive name (e.g., "parallel", "for")
161pub fn lex_directive(input: &str) -> IResult<&str, &str> {
162    lex_identifier(input)
163}
164
165/// Parse a clause name (e.g., "private", "nowait")
166pub fn lex_clause(input: &str) -> IResult<&str, &str> {
167    lex_identifier(input)
168}

170/// Skip whitespace and C-style comments
171///
172/// Learning Rust: Manual Parsing and Byte Manipulation
173/// ====================================================
174/// Sometimes you need to go beyond parser combinators!
175/// This function manually iterates through bytes for performance
176pub fn skip_space_and_comments(input: &str) -> IResult<&str, &str> {
177    let mut i = 0;
178    let bytes = input.as_bytes();
179    let len = bytes.len();
180
181    while i < len {
182        // Learning Rust: Working with Bytes
183        // ==================================
184        // .as_bytes() converts &str to &[u8] (byte slice)
185        // Useful for ASCII operations (faster than chars)
186        if bytes[i].is_ascii_whitespace() {
187            // Learning Rust: UTF-8 Handling
188            // ==============================
189            // chars() iterates over Unicode scalar values
190            // len_utf8() returns bytes needed for this character
191            let ch = input[i..].chars().next().unwrap();
192            i += ch.len_utf8();
193            continue;
194        }
195
196        // Handle /* */ comments
197        if i + 1 < len && &input[i..i + 2] == "/*" {
198            if let Some(end) = input[i + 2..].find("*/") {
199                i += 2 + end + 2;
200                continue;
201            } else {
202                // Unterminated comment - consume to end
203                i = len;
204                break;
205            }
206        }
207
208        // Handle // comments
209        if i + 1 < len && &input[i..i + 2] == "//" {
210            if let Some(end) = input[i + 2..].find('\n') {
211                i += 2 + end + 1;
212            } else {
213                i = len;
214            }
215            continue;
216        }
217
218        break;
219    }
220
221    // Return (remaining, consumed) - consumed is empty slice for compatibility
222    Ok((&input[i..], &input[..0]))
223}

225/// Skip whitespace/comments - requires at least one
226pub fn skip_space1_and_comments(input: &str) -> IResult<&str, &str> {
227    let (rest, _) = skip_space_and_comments(input)?;
228
229    // Learning Rust: Error Handling in Parsers
230    // =========================================
231    // Return an error if nothing was consumed
232    if rest.len() == input.len() {
233        Err(nom::Err::Error(nom::error::Error::new(
234            input,
235            nom::error::ErrorKind::Space,
236        )))
237    } else {
238        Ok((rest, &input[..0]))
239    }
240}
