1//! Language-aware helpers for parsing clause items.
2//!
3//! The core IR conversion logic treats clause payloads generically.  This
4//! module provides just enough C/C++/Fortran awareness to interpret
5//! OpenMP-specific constructs such as array sections while keeping the
6//! feature easy to disable via [`ParserConfig`].
7
8use super::{
9    ArraySection, ClauseItem, ConversionError, Expression, Identifier, Language, ParserConfig,
10    Variable,

13/// Parse a comma separated list of clause items using language aware rules.
14pub fn parse_clause_item_list(
15    content: &str,
16    config: &ParserConfig,
17) -> Result<Vec<ClauseItem>, ConversionError> {
18    if !config.language_semantics_enabled() {
19        return Ok(fallback_identifier_list(content));
20    }
21
22    let language = config.language();
23    let segments = split_top_level(content, ',', &[('[', ']'), ('(', ')')]);
24    let mut items = Vec::new();
25
26    for raw in segments {
27        let trimmed = raw.trim();
28        if trimmed.is_empty() {
29            continue;
30        }
31
32        // If it contains a top-level '[' then treat as C-like variable with array sections
33        // Detect C-like array sections by splitting on top-level '[' (pairs empty
34        // so '[' is recognized unless inside quotes/templates).
35        if split_top_level(trimmed, '[', &[] as &[(char, char)]).len() > 1 {
36            let variable = parse_c_like_variable(trimmed, config)?;
37            items.push(ClauseItem::Variable(variable));
38            continue;
39        }
40
41        // Fortran form: name(...) with parentheses. Use a simple contains +
42        // ends_with check and let parse_fortran_variable handle nesting properly.
43        if matches!(language, Language::Fortran) && trimmed.contains('(') && trimmed.ends_with(')')
44        {
45            if let Some(variable) = parse_fortran_variable(trimmed, config)? {
46                items.push(ClauseItem::Variable(variable));
47                continue;
48            }
49        }
50
51        if looks_like_identifier(trimmed) {
52            items.push(ClauseItem::Identifier(Identifier::new(trimmed)));
53            continue;
54        }
55
56        items.push(ClauseItem::Expression(Expression::new(trimmed, config)));
57    }
58
59    Ok(items)
60}

62fn fallback_identifier_list(content: &str) -> Vec<ClauseItem> {
63    content
64        .split(',')
65        .map(|s| s.trim())
66        .filter(|s| !s.is_empty())
67        .map(|s| ClauseItem::Identifier(Identifier::new(s)))
68        .collect()
69}

71fn parse_c_like_variable(value: &str, config: &ParserConfig) -> Result<Variable, ConversionError> {
72    let name_end = value.find('[').ok_or_else(|| {
73        ConversionError::InvalidClauseSyntax(format!("expected array designator in `{value}`"))
74    })?;
75
76    let name = value[..name_end].trim();
77    if name.is_empty() {
78        return Err(ConversionError::InvalidClauseSyntax(format!(
79            "missing variable name before array section in `{value}`"
80        )));
81    }
82
83    let mut sections = Vec::new();
84    let mut rest = &value[name_end..];
85    while rest.starts_with('[') {
86        let (content, remaining) = extract_bracket_content(rest, '[', ']')?;
87        let section = parse_array_section(content, config, false)?; // C uses lower:length
88        sections.push(section);
89        rest = remaining.trim_start();
90    }
91
92    if !rest.trim().is_empty() {
93        return Err(ConversionError::InvalidClauseSyntax(format!(
94            "unexpected trailing characters `{}` in `{}`",
95            rest.trim(),
96            value
97        )));
98    }
99
100    Ok(Variable::with_sections(name, sections))
101}

103fn parse_fortran_variable(
104    value: &str,
105    config: &ParserConfig,
106) -> Result<Option<Variable>, ConversionError> {
107    let trimmed = value.trim();
108    let Some(paren_pos) = trimmed.find('(') else {
109        return Ok(None);
110    };
111
112    let name = trimmed[..paren_pos].trim();
113    if name.is_empty() {
114        return Err(ConversionError::InvalidClauseSyntax(format!(
115            "missing variable name before parenthesis in `{value}`"
116        )));
117    }
118
119    let section_part = &trimmed[paren_pos..];
120    let (content, remainder) = extract_bracket_content(section_part, '(', ')')?;
121    if !remainder.trim().is_empty() {
122        return Err(ConversionError::InvalidClauseSyntax(format!(
123            "unexpected trailing characters `{}` in `{}`",
124            remainder.trim(),
125            value
126        )));
127    }
128
129    let mut sections = Vec::new();
130    if !content.trim().is_empty() {
131        for dimension in split_top_level(content, ',', &[('(', ')'), ('[', ']')]) {
132            let section = parse_array_section(dimension, config, true)?; // Fortran uses lower:upper
133            sections.push(section);
134        }
135    }
136
137    if sections.is_empty() {
138        return Ok(Some(Variable::new(name)));
139    }
140
141    Ok(Some(Variable::with_sections(name, sections)))
142}

144fn parse_array_section(
145    value: &str,
146    config: &ParserConfig,
147    is_fortran: bool,
148) -> Result<ArraySection, ConversionError> {
149    let trimmed = value.trim();
150
151    if trimmed.is_empty() {
152        return Ok(ArraySection::all());
153    }
154
155    let parts = split_top_level(trimmed, ':', &[('(', ')'), ('[', ']')]);
156    match parts.len() {
157        1 => {
158            let expr = Expression::new(parts[0], config);
159            Ok(ArraySection::single_index(expr))
160        }
161        2 => {
162            let lower = if parts[0].is_empty() {
163                None
164            } else {
165                Some(Expression::new(parts[0], config))
166            };
167
168            // Fortran uses lower:upper, C uses lower:length
169            let length = if parts[1].is_empty() {
170                None
171            } else if is_fortran {
172                // Convert Fortran upper bound to length: length = upper - lower + 1
173                let upper = Expression::new(parts[1], config);
174                Some(fortran_upper_to_length(lower.as_ref(), upper, None, config))
175            } else {
176                Some(Expression::new(parts[1], config))
177            };
178            Ok(ArraySection::new(lower, length, None))
179        }
180        3 => {
181            let lower = if parts[0].is_empty() {
182                None
183            } else {
184                Some(Expression::new(parts[0], config))
185            };
186
187            let stride = if parts[2].is_empty() {
188                None
189            } else {
190                Some(Expression::new(parts[2], config))
191            };
192
193            // Fortran uses lower:upper:stride, C uses lower:length:stride
194            let length = if parts[1].is_empty() {
195                None
196            } else if is_fortran {
197                // Convert Fortran upper bound to length with stride: ((upper - lower) / stride) + 1
198                let upper = Expression::new(parts[1], config);
199                Some(fortran_upper_to_length(
200                    lower.as_ref(),
201                    upper,
202                    stride.as_ref(),
203                    config,
204                ))
205            } else {
206                Some(Expression::new(parts[1], config))
207            };
208
209            Ok(ArraySection::new(lower, length, stride))
210        }
211        _ => Err(ConversionError::InvalidClauseSyntax(format!(
212            "too many ':' separators in array section `{value}`"
213        ))),
214    }
215}

217/// Convert Fortran upper bound to length expression with optional stride
218///
219/// Fortran array sections use lower:upper[:stride] syntax where both bounds are inclusive.
220/// ArraySection stores length, so we convert:
221/// - Without stride: length = upper - lower + 1
222/// - With stride: length = ((upper - lower) / stride) + 1
223///
224/// # Examples
225///
226/// - `A(5:10)` → lower=5, upper=10 → length = (10 - 5 + 1) = 6 elements
227/// - `A(:10)` → lower=implicit 1, upper=10 → length = 10 elements
228/// - `A(1:N)` → lower=1, upper=N → length = N elements
229/// - `A(1:10:2)` → lower=1, upper=10, stride=2 → length = ((10 - 1) / 2) + 1 = 5 elements
230fn fortran_upper_to_length(
231    lower: Option<&Expression>,
232    upper: Expression,
233    stride: Option<&Expression>,
234    config: &ParserConfig,
235) -> Expression {
236    match (lower, stride) {
237        (Some(lower_expr), Some(stride_expr)) => {
238            // With explicit lower and stride: ((upper - lower) / stride) + 1
239            Expression::new(
240                format!(
241                    "((({}) - ({})) / ({}) + 1)",
242                    upper.as_str(),
243                    lower_expr.as_str(),
244                    stride_expr.as_str()
245                ),
246                config,
247            )
248        }
249        (Some(lower_expr), None) => {
250            // With explicit lower, no stride: (upper - lower + 1)
251            Expression::new(
252                format!("(({}) - ({}) + 1)", upper.as_str(), lower_expr.as_str()),
253                config,
254            )
255        }
256        (None, Some(stride_expr)) => {
257            // No lower (implicit 1), with stride: ((upper - 1) / stride) + 1
258            Expression::new(
259                format!(
260                    "((({}) - 1) / ({}) + 1)",
261                    upper.as_str(),
262                    stride_expr.as_str()
263                ),
264                config,
265            )
266        }
267        (None, None) => {
268            // No lower (implicit 1), no stride: upper - 1 + 1 = upper
269            upper
270        }
271    }
272}

274fn split_top_level<'a>(input: &'a str, separator: char, pairs: &[(char, char)]) -> Vec<&'a str> {
275    let mut segments = Vec::new();
276    let mut start = 0;
277    let mut stack: Vec<char> = Vec::new();
278    let mut in_single_quote = false;
279    let mut in_double_quote = false;
280    let mut angle_depth = 0usize;
281
282    let chars: Vec<(usize, char)> = input.char_indices().collect();
283    let mut i = 0;
284
285    while i < chars.len() {
286        let (idx, ch) = chars[i];
287        let prev = if i > 0 { Some(chars[i - 1].1) } else { None };
288        let next = chars.get(i + 1).map(|(_, c)| *c);
289
290        // Track quote state
291        match ch {
292            '\'' if !in_double_quote => {
293                in_single_quote = !in_single_quote;
294            }
295            '"' if !in_single_quote => {
296                in_double_quote = !in_double_quote;
297            }
298            _ if in_single_quote || in_double_quote => {
299                // Inside quotes, skip delimiter checking
300            }
301            '<' if stack.is_empty() => {
302                // Detect C++ templates
303                if is_template_start(prev, next) {
304                    angle_depth += 1;
305                }
306            }
307            '>' if angle_depth > 0 && stack.is_empty() => {
308                // Close template - handle >> as two closings for nested templates
309                // Skip only if this is part of >>= or >= (comparison operators)
310                if matches!(next, Some('=')) && !matches!(prev, Some('>')) {
311                    // This is >= comparison, skip it
312                } else {
313                    // This is either a single > or part of >> (nested template closing)
314                    // Both should decrement angle_depth
315                    angle_depth = angle_depth.saturating_sub(1);
316                }
317            }
318            _ if !in_single_quote && !in_double_quote => {
319                // Track paired delimiters
320                if let Some(&(_, close)) = pairs.iter().find(|(open, _)| *open == ch) {
321                    stack.push(close);
322                } else if stack.last().copied() == Some(ch) {
323                    stack.pop();
324                }
325            }
326            _ => {}
327        }
328
329        // Split at separator only if not inside any structure
330        if ch == separator
331            && stack.is_empty()
332            && angle_depth == 0
333            && !in_single_quote
334            && !in_double_quote
335        {
336            segments.push(input[start..idx].trim());
337            start = idx + ch.len_utf8();
338        }
339
340        i += 1;
341    }
342
343    segments.push(input[start..].trim());
344    segments
345}

347/// Check if '<' is likely a template start rather than comparison operator.
348fn is_template_start(prev: Option<char>, next: Option<char>) -> bool {
349    let is_prev_valid =
350        matches!(prev, Some(c) if c.is_alphanumeric() || c == '_' || c == ':' || c == '>');
351    let is_next_disqualifier = matches!(next, Some('=') | Some('<') | Some('>'));
352    is_prev_valid && !is_next_disqualifier
353}

355/// Split the input once at the first top-level occurrence of `delimiter`.
356///
357/// Returns None if the delimiter is not found at the top level.
358pub fn split_once_top_level(input: &str, delimiter: char) -> Option<(&str, &str)> {
359    find_top_level_delimiter(input, delimiter, false).map(|idx| {
360        let next = idx + delimiter.len_utf8();
361        (&input[..idx], &input[next..])
362    })
363}

365/// Split the input once at the last top-level occurrence of `delimiter`.
366///
367/// Returns None if the delimiter is not found at the top level.
368pub fn rsplit_once_top_level(input: &str, delimiter: char) -> Option<(&str, &str)> {
369    find_top_level_delimiter(input, delimiter, true).map(|idx| {
370        let next = idx + delimiter.len_utf8();
371        (&input[..idx], &input[next..])
372    })
373}

375/// Find the first (or last) top-level occurrence of a delimiter.
376///
377/// Respects nesting of parentheses, brackets, braces, and quotes.
378/// For colon delimiter, also handles ternary operator (? :) disambiguation
379/// and C++ scope operator (::) to avoid false matches.
380fn find_top_level_delimiter(input: &str, delimiter: char, from_end: bool) -> Option<usize> {
381    let mut depth_paren = 0i32;
382    let mut depth_bracket = 0i32;
383    let mut depth_brace = 0i32;
384    let mut ternary_depth = 0i32;
385    let mut in_single_quote = false;
386    let mut in_double_quote = false;
387    let mut result: Option<usize> = None;
388
389    let chars: Vec<(usize, char)> = input.char_indices().collect();
390
391    for i in 0..chars.len() {
392        let (idx, ch) = chars[i];
393        let next_ch = chars.get(i + 1).map(|(_, c)| *c);
394        let prev_ch = if i > 0 { Some(chars[i - 1].1) } else { None };
395
396        // Track quote state
397        match ch {
398            '\'' if !in_double_quote => {
399                in_single_quote = !in_single_quote;
400                continue;
401            }
402            '"' if !in_single_quote => {
403                in_double_quote = !in_double_quote;
404                continue;
405            }
406            _ if in_single_quote || in_double_quote => {
407                continue;
408            }
409            _ => {}
410        }
411
412        // Track delimiter depth
413        match ch {
414            '(' => depth_paren += 1,
415            ')' => {
416                if depth_paren > 0 {
417                    depth_paren -= 1;
418                }
419            }
420            '[' => depth_bracket += 1,
421            ']' => {
422                if depth_bracket > 0 {
423                    depth_bracket -= 1;
424                }
425            }
426            '{' => depth_brace += 1,
427            '}' => {
428                if depth_brace > 0 {
429                    depth_brace -= 1;
430                }
431            }
432            '?' if delimiter == ':' => ternary_depth += 1,
433            ':' if delimiter == ':' && ternary_depth > 0 => {
434                ternary_depth -= 1;
435                continue;
436            }
437            _ => {}
438        }
439
440        if ch == delimiter && depth_paren == 0 && depth_bracket == 0 && depth_brace == 0 {
441            // Skip C++ scope operator :: when delimiter is :
442            if delimiter == ':' && (prev_ch == Some(':') || next_ch == Some(':')) {
443                continue;
444            }
445
446            result = Some(idx);
447            if !from_end {
448                break;
449            }
450        }
451    }
452
453    result
454}

456/// Extract content from a bracketed section, handling nesting.
457///
458/// Returns (content, remainder) where content is what's inside the delimiters.
459/// Supports any pair of delimiters: `()`, `[]`, `{}`, etc.
460///
461/// This is used by both the lang module and convert.rs to avoid duplication.
462pub(crate) fn extract_bracket_content(
463    input: &str,
464    open: char,
465    close: char,
466) -> Result<(&str, &str), ConversionError> {
467    if !input.starts_with(open) {
468        return Err(ConversionError::InvalidClauseSyntax(format!(
469            "expected `{input}` to start with '{open}'"
470        )));
471    }
472
473    let mut depth = 0;
474    let mut start_idx = None;
475    for (idx, ch) in input.char_indices() {
476        if ch == open {
477            depth += 1;
478            if depth == 1 {
479                start_idx = Some(idx + open.len_utf8());
480            }
481        } else if ch == close {
482            depth -= 1;
483            if depth == 0 {
484                let start = start_idx.expect("opening bracket must set start index");
485                let content = &input[start..idx];
486                let rest = &input[idx + close.len_utf8()..];
487                return Ok((content, rest));
488            }
489        }
490    }
491
492    Err(ConversionError::InvalidClauseSyntax(format!(
493        "unterminated `{open}` block in `{input}`"
494    )))
495}

497fn looks_like_identifier(value: &str) -> bool {
498    let mut chars = value.chars();
499    match chars.next() {
500        Some(ch) if is_identifier_start(ch) => chars.all(is_identifier_continue),
501        _ => false,
502    }
503}
504
505fn is_identifier_start(ch: char) -> bool {
506    ch.is_alphabetic() || ch == '_'
507}
508
509fn is_identifier_continue(ch: char) -> bool {
510    ch.is_alphanumeric() || ch == '_'
511}
