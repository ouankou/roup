323pub fn clause_registry() -> ClauseRegistry {
324    let mut builder = ClauseRegistryBuilder::new().with_default_rule(OPENMP_DEFAULT_CLAUSE_RULE);
325
326    for clause in OpenMpClause::ALL {
327        builder.register_with_rule_mut(clause.name(), clause.rule());
328    }
329
330    builder.build()
331}

333// Helper function to parse balanced parentheses and extract content
334fn parse_parenthesized_content(input: &str) -> nom::IResult<&str, String> {
335    use crate::lexer;
336    use nom::bytes::complete::tag;
337    use nom::error::{Error, ErrorKind};
338
339    // Skip whitespace and comments before the opening parenthesis
340    let (input, _) = lexer::skip_space_and_comments(input)?;
341
342    // Expect an opening parenthesis
343    let (input, _) = tag("(")(input)?;
344
345    // Find the matching closing parenthesis, tracking depth
346    let mut depth = 1;
347    let mut end_index = None;
348
349    for (idx, ch) in input.char_indices() {
350        match ch {
351            '(' => depth += 1,
352            ')' => {
353                depth -= 1;
354                if depth == 0 {
355                    end_index = Some(idx);
356                    break;
357                }
358            }
359            _ => {}
360        }
361    }
362
363    let end_index = end_index.ok_or_else(|| nom::Err::Error(Error::new(input, ErrorKind::Fail)))?;
364
365    let raw_content = &input[..end_index];
366    let trimmed = raw_content.trim();
367    let normalized = lexer::collapse_line_continuations(trimmed);
368    let rest = &input[end_index + 1..];
369
370    Ok((rest, normalized.into_owned()))
371}

373// Custom parser for allocate directive: allocate(list) [clauses] or bare allocate
374fn parse_allocate_directive<'a>(
375    name: std::borrow::Cow<'a, str>,
376    input: &'a str,
377    clause_registry: &ClauseRegistry,
378) -> nom::IResult<&'a str, super::Directive<'a>> {
379    use super::Directive;
380
381    // Try to parse parenthesized list (extended form)
382    if let Ok((rest, list_content)) = parse_parenthesized_content(input) {
383        let (rest, clauses) = clause_registry.parse_sequence(rest)?;
384
385        Ok((
386            rest,
387            Directive {
388                name: std::borrow::Cow::Borrowed("allocate"),
389                parameter: Some(std::borrow::Cow::Owned(format!("({})", list_content))),
390                clauses,
391            },
392        ))
393    } else {
394        // Fall back to standard clause parsing (bare form)
395        let (rest, clauses) = clause_registry.parse_sequence(input)?;
396        Ok((
397            rest,
398            Directive {
399                name,
400                parameter: None,
401                clauses,
402            },
403        ))
404    }
405}

557                parameter: Some(std::borrow::Cow::Owned(format!("({})", depobj_id))),
558                clauses,
559            },
560        ))
561    } else {
562        // Fall back to standard clause parsing (bare form)
563        let (rest, clauses) = clause_registry.parse_sequence(input)?;
564        Ok((
565            rest,
566            Directive {
567                name,
568                parameter: None,
569                clauses,
570            },
571        ))
572    }
573}

575// Custom parser for scan directive: scan exclusive(list) or scan inclusive(list) or bare scan
576fn parse_scan_directive<'a>(
577    name: std::borrow::Cow<'a, str>,
578    input: &'a str,
579    clause_registry: &ClauseRegistry,
580) -> nom::IResult<&'a str, super::Directive<'a>> {
581    use super::Directive;
582    use nom::bytes::complete::tag;
583
584    // Trim leading whitespace for reliable token matching
585    let input_trimmed = input.trim_start();
586
587    // Try exclusive first: `exclusive(list)` optionally followed by clauses
588    if let Ok((rest_after_tag, _)) =
589        tag::<_, _, nom::error::Error<&str>>("exclusive")(input_trimmed)
590    {
591        if let Ok((rest_after_paren, list_content)) = parse_parenthesized_content(rest_after_tag) {
592            // After the parenthesized list we may have clauses. Parse them from the remaining input.
593            let (rest, clauses) = match clause_registry.parse_sequence(rest_after_paren) {
594                Ok((r, c)) => (r, c),
595                Err(e) => {
596                    // Distinguish between: (A) Truly no clauses (only whitespace/comments remaining)
597                    // and (B) a parse error. If the remainder contains only whitespace/comments,
598                    // treat as no clauses; otherwise propagate the parse error so malformed
599                    // clauses aren't silently accepted.
600                    if let Ok((remaining_after_skipping, _)) =
601                        crate::lexer::skip_space_and_comments(rest_after_paren)
602                    {
603                        if remaining_after_skipping.is_empty() {
604                            (rest_after_paren, vec![])
605                        } else {
606                            return Err(e);
607                        }
608                    } else {
609                        return Err(e);
610                    }
611                }
612            };
613
614            return Ok((
615                rest,
616                Directive {
617                    name: std::borrow::Cow::Borrowed("scan"),
618                    // Store the parameter without a leading space; the display/formatting layer
619                    // is responsible for spacing when rendering the directive.
620                    parameter: Some(std::borrow::Cow::Owned(format!(
621                        "exclusive({})",
622                        list_content
623                    ))),
624                    clauses,
625                },
626            ));
627        }
628    }
629
630    // Try inclusive: `inclusive(list)` optionally followed by clauses
631    if let Ok((rest_after_tag, _)) =
632        tag::<_, _, nom::error::Error<&str>>("inclusive")(input_trimmed)
633    {
634        if let Ok((rest_after_paren, list_content)) = parse_parenthesized_content(rest_after_tag) {
635            let (rest, clauses) = match clause_registry.parse_sequence(rest_after_paren) {
636                Ok((r, c)) => (r, c),
637                Err(e) => {
638                    if let Ok((remaining_after_skipping, _)) =
639                        crate::lexer::skip_space_and_comments(rest_after_paren)
640                    {
641                        if remaining_after_skipping.is_empty() {
642                            (rest_after_paren, vec![])
643                        } else {
644                            return Err(e);
645                        }
646                    } else {
647                        return Err(e);
648                    }
649                }
650            };
651
652            return Ok((
653                rest,
654                Directive {
655                    name: std::borrow::Cow::Borrowed("scan"),
656                    parameter: Some(std::borrow::Cow::Owned(format!(
657                        "inclusive({})",
658                        list_content
659                    ))),
660                    clauses,
661                },
662            ));
663        }
664    }
665
666    // Fall back to standard clause parsing (bare form)
667    let (rest, clauses) = clause_registry.parse_sequence(input)?;
668    Ok((
669        rest,
670        Directive {
671            name,
672            parameter: None,
673            clauses,
674        },
675    ))
676}

701                clauses,
702            },
703        ))
704    } else {
705        // Fall back to standard clause parsing (bare form)

707        Ok((
708            rest,
709            Directive {
710                name,
711                parameter: None,
712                clauses,
713            },
714        ))
715    }

773// Directive names that have custom parsers (excluding target_data underscore variant)
774const CUSTOM_PARSER_DIRECTIVES: &[&str] = &[
775    "allocate",
776    "threadprivate",
777    "declare target",
778    "declare mapper",
779    "declare variant",
780    "depobj",
781    "scan",
782    "cancel",
783    "groupprivate",
784];
785
786pub fn directive_registry() -> DirectiveRegistry {
787    let mut builder = DirectiveRegistryBuilder::new();
788
789    // Register custom parsers for directives with special syntax
790    builder = builder.register_custom("allocate", parse_allocate_directive);
791    builder = builder.register_custom("threadprivate", parse_threadprivate_directive);
792    builder = builder.register_custom("declare target", parse_declare_target_extended);
793    builder = builder.register_custom("declare mapper", parse_declare_mapper_directive);
794    builder = builder.register_custom("declare variant", parse_declare_variant_directive);
795    builder = builder.register_custom("depobj", parse_depobj_directive);
796    builder = builder.register_custom("scan", parse_scan_directive);
797    builder = builder.register_custom("cancel", parse_cancel_directive);
798    builder = builder.register_custom("groupprivate", parse_groupprivate_directive);
799
800    // Handle underscore variant of "target data"
801    builder = builder.register_custom("target_data", parse_target_data_directive);
802
803    // Register remaining directives as generic
804    for directive in OpenMpDirective::ALL {
805        let name = directive.as_str();
806        // Skip directives that already have custom parsers
807        if !CUSTOM_PARSER_DIRECTIVES.contains(&name) {
808            builder = builder.register_generic(name);
809        }
810    }
811
812    builder.build()
813}
