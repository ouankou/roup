319pub fn clause_registry() -> ClauseRegistry {
320    let mut builder = ClauseRegistryBuilder::new().with_default_rule(OPENMP_DEFAULT_CLAUSE_RULE);
321
322    for clause in OpenMpClause::ALL {
323        builder.register_with_rule_mut(clause.name(), clause.rule());
324    }
325
326    builder.build()
327}
328
329pub fn directive_registry() -> DirectiveRegistry {
330    let mut builder = DirectiveRegistryBuilder::new();
331
332    for directive in OpenMpDirective::ALL {
333        builder = builder.register_generic(directive.as_str());
334    }
335
336    builder.build()
337}
338
339pub fn parser() -> Parser {
340    Parser::new(directive_registry(), clause_registry())
