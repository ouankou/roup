138pub fn clause_registry() -> ClauseRegistry {
139    let mut builder = ClauseRegistryBuilder::new().with_default_rule(OPENACC_DEFAULT_CLAUSE_RULE);
140
141    for clause in OpenAccClause::ALL {
142        // Skip clauses that will have custom parsers
143        let name = clause.name();
144        if matches!(
145            name,
146            "copyin" | "pcopyin" | "present_or_copyin" |
147            "copyout" | "pcopyout" | "present_or_copyout" |
148            "create" | "pcreate" | "present_or_create" |
149            "reduction" |
150            // Variable list clauses (need custom parsing to split comma-separated vars)
151            "gang" | "worker" | "vector" | "wait" | "private" | "firstprivate" |
152            "device_type" | "dtype" |
153            "copy" | "pcopy" | "present_or_copy" | "present" | "attach" | "detach" |
154            "use_device" | "link" | "device_resident" | "host" | "device" |
155            "deviceptr" | "delete"
156        ) {
157            continue;
158        }
159        builder.register_with_rule_mut(name, clause.rule());
160    }
161
162    // Register custom parsers for clauses with modifiers/operators
163    builder.register_with_rule_mut("copyin", ClauseRule::Custom(parse_copyin_clause));
164    builder.register_with_rule_mut("pcopyin", ClauseRule::Custom(parse_copyin_clause));
165    builder.register_with_rule_mut("present_or_copyin", ClauseRule::Custom(parse_copyin_clause));
166
167    builder.register_with_rule_mut("copyout", ClauseRule::Custom(parse_copyout_clause));
168    builder.register_with_rule_mut("pcopyout", ClauseRule::Custom(parse_copyout_clause));
169    builder.register_with_rule_mut(
170        "present_or_copyout",
171        ClauseRule::Custom(parse_copyout_clause),
172    );
173
174    builder.register_with_rule_mut("create", ClauseRule::Custom(parse_create_clause));
175    builder.register_with_rule_mut("pcreate", ClauseRule::Custom(parse_create_clause));
176    builder.register_with_rule_mut("present_or_create", ClauseRule::Custom(parse_create_clause));
177
178    builder.register_with_rule_mut("reduction", ClauseRule::Custom(parse_reduction_clause));
179
180    // Register specialized parsers for gang/worker/vector with modifier support
181    builder.register_with_rule_mut("gang", ClauseRule::Custom(parse_gang_clause));
182    builder.register_with_rule_mut("worker", ClauseRule::Custom(parse_worker_clause));
183    builder.register_with_rule_mut("vector", ClauseRule::Custom(parse_vector_clause));
184
185    // Register variable list parsers for clauses that take comma-separated variable lists
186    // These need custom parsing to split "a, b, c" into ["a", "b", "c"] for proper deduplication
187    // NOTE: tile, collapse, num_gangs, and async are NOT included here because they take
188    // parameter lists (literal values), not variable lists, and should not be merged
189    let var_list_clauses = [
190        "wait",
191        "private",
192        "firstprivate",
193        "device_type",
194        "dtype",
195        "copy",
196        "pcopy",
197        "present_or_copy",
198        "present",
199        "attach",
200        "detach",
201        "use_device",
202        "link",
203        "device_resident",
204        "host",
205        "device",
206        "deviceptr",
207        "delete",
208    ];
209    for clause_name in &var_list_clauses {
210        builder.register_with_rule_mut(clause_name, ClauseRule::Custom(parse_variable_list_clause));
211    }
212
213    builder.build()
214}

216fn parse_cache_directive<'a>(
217    name: Cow<'a, str>,
218    input: &'a str,
219    clause_registry: &ClauseRegistry,
220) -> nom::IResult<&'a str, super::Directive<'a>> {
221    use super::{CacheDirectiveData, Directive};
222    use crate::lexer;
223
224    let (input, _) = lexer::skip_space_and_comments(input)?;
225    let (rest_after_paren, content) = parse_parenthesized_content_inner(input)?;
226    let (rest, clauses) = clause_registry.parse_sequence(rest_after_paren)?;
227
228    // Parse cache directive: cache(readonly: x, y, z) or cache(x, y, z)
229    let content_trimmed = content.trim();
230    let (readonly, var_part) = if let Some(stripped) = content_trimmed.strip_prefix("readonly:") {
231        (true, stripped.trim())
232    } else if let Some(stripped) = content_trimmed.strip_prefix("readonly :") {
233        (true, stripped.trim())
234    } else {
235        (false, content_trimmed)
236    };
237
238    // Parse variable list (parse_variable_list returns owned Cow values)
239    let variables: Vec<Cow<'a, str>> = parse_variable_list(var_part)
240        .into_iter()
241        .map(|v| Cow::Owned(v.into_owned()))
242        .collect();
243
244    // For backward compat: keep normalized parameter
245    let normalized = normalize_directive_parameter(content.trim());
246    let parameter = format!("({})", normalized);
247
248    Ok((
249        rest,
250        Directive {
251            name,
252            parameter: Some(Cow::Owned(parameter)),
253            clauses,
254            wait_data: None,
255            cache_data: Some(CacheDirectiveData {
256                readonly,
257                variables,
258            }),
259        },
260    ))
261}

263/// Normalize whitespace in wait/cache directive parameters
264/// Format as "keyword: value": "devnum : 23 : queues: 1, 2" -> "devnum: 23: queues: 1, 2"
265fn normalize_directive_parameter(s: &str) -> String {
266    let mut result = String::with_capacity(s.len() + 4);
267    let mut prev_char = ' ';
268    let mut after_colon = false;
269
270    for ch in s.chars() {
271        if ch == ':' {
272            // Remove trailing spaces before colon
273            while result.ends_with(' ') {
274                result.pop();
275            }
276            result.push(':');
277            result.push(' '); // Always add one space after colon
278            after_colon = true;
279            prev_char = ':';
280        } else if ch == ' ' || ch == '\t' {
281            if after_colon {
282                // Skip leading spaces after colon (we already added one)
283                continue;
284            }
285            if prev_char != ' ' {
286                result.push(' ');
287                prev_char = ' ';
288            }
289        } else {
290            after_colon = false;
291            result.push(ch);
292            prev_char = ch;
293        }
294    }
295
296    result.trim().to_string()
297}

299fn parse_wait_directive<'a>(
300    name: Cow<'a, str>,
301    input: &'a str,
302    clause_registry: &ClauseRegistry,
303) -> nom::IResult<&'a str, super::Directive<'a>> {
304    use super::{Directive, WaitDirectiveData};
305    use crate::lexer;
306
307    let (input, _) = lexer::skip_space_and_comments(input)?;
308    if input.trim_start().starts_with('(') {
309        let (rest, content) = parse_parenthesized_content_inner(input)?;
310        let (rest, clauses) = clause_registry.parse_sequence(rest)?;
311
312        // Parse wait directive: wait(devnum: 23: 1, 2, 3) or wait(devnum: 23: queues: 1, 2, 3) or wait(1, 2, 3)
313        let content_str = content.as_str().trim();
314        let mut devnum = None;
315        let mut has_queues = false;
316        let mut queue_exprs = Vec::new();
317
318        if let Some(stripped) = content_str
319            .strip_prefix("devnum:")
320            .or_else(|| content_str.strip_prefix("devnum :"))
321        {
322            // Parse: "devnum: 23" or "devnum: 23: queues: ..."
323            let after_devnum = stripped.trim();
324
325            if let Some(colon_pos) = after_devnum.find(':') {
326                // Has queue list or additional content after devnum
327                devnum = Some(Cow::Owned(after_devnum[..colon_pos].trim().to_string()));
328                let rest = after_devnum[colon_pos + 1..].trim();
329
330                if let Some(queues_stripped) = rest
331                    .strip_prefix("queues:")
332                    .or_else(|| rest.strip_prefix("queues :"))
333                {
334                    has_queues = true;
335                    queue_exprs = parse_variable_list(queues_stripped.trim());
336                } else {
337                    queue_exprs = parse_variable_list(rest);
338                }
339            } else {
340                // Only devnum, no queue list
341                devnum = Some(Cow::Owned(after_devnum.to_string()));
342            }
343        } else if let Some(stripped) = content_str.strip_prefix("queues:") {
344            // Parse: "queues: 1, 2, 3"
345            has_queues = true;
346            queue_exprs = parse_variable_list(stripped.trim());
347        } else if let Some(stripped) = content_str.strip_prefix("queues :") {
348            // Parse: "queues : 1, 2, 3"
349            has_queues = true;
350            queue_exprs = parse_variable_list(stripped.trim());
351        } else if !content_str.is_empty() {
352            // Parse: "1, 2, 3"
353            queue_exprs = parse_variable_list(content_str);
354        }
355
356        // Ensure all queue_exprs are owned (not borrowed from content)
357        let queue_exprs_owned: Vec<Cow<'a, str>> = queue_exprs
358            .into_iter()
359            .map(|v| Cow::Owned(v.into_owned()))
360            .collect();
361
362        // For backward compat: keep normalized parameter
363        let normalized = normalize_directive_parameter(content.trim());
364        let parameter = format!("({})", normalized);
365
366        return Ok((
367            rest,
368            Directive {
369                name,
370                parameter: Some(Cow::Owned(parameter)),
371                clauses,
372                wait_data: Some(WaitDirectiveData {
373                    devnum,
374                    has_queues,
375                    queue_exprs: queue_exprs_owned,
376                }),
377                cache_data: None,
378            },
379        ));
380    }
381
382    let (rest, clauses) = clause_registry.parse_sequence(input)?;
383    Ok((
384        rest,
385        Directive {
386            name,
387            parameter: None,
388            clauses,
389            wait_data: None,
390            cache_data: None,
391        },
392    ))
393}

395fn parse_end_directive<'a>(
396    name: Cow<'a, str>,
397    input: &'a str,
398    clause_registry: &ClauseRegistry,
399) -> nom::IResult<&'a str, super::Directive<'a>> {
400    use super::Directive;
401    use crate::lexer;
402
403    let (mut rest, _) = lexer::skip_space_and_comments(input)?;
404
405    // FIX: Read full multi-word directive name (e.g., "parallel loop") instead of just one token
406    // Keep reading identifiers until we hit a non-identifier
407    let mut directive_parts = Vec::new();
408    while let Ok((new_rest, token)) = lexer::lex_identifier_token(rest) {
409        let collapsed = lexer::collapse_line_continuations(token);
410        directive_parts.push(collapsed.as_ref().to_string());
411        rest = new_rest;
412
413        // Skip whitespace and try again
414        if let Ok((new_rest, _)) = lexer::skip_space_and_comments(rest) {
415            rest = new_rest;
416        }
417    }
418
419    // Store the directive being ended as a parameter
420    let directive = directive_parts.join(" ");
421    let (rest, clauses) = clause_registry.parse_sequence(rest)?;
422    // Store the directive token without presentation spacing; rendering should add spaces.
423    let parameter = directive.to_string();
424
425    Ok((
426        rest,
427        Directive {
428            name,
429            parameter: Some(Cow::Owned(parameter)),
430            clauses,
431            wait_data: None,
432            cache_data: None,
433        },
434    ))
435}

476pub fn directive_registry() -> DirectiveRegistry {
477    let mut builder = DirectiveRegistryBuilder::new();
478
479    builder = builder.register_custom("cache", parse_cache_directive);
480    builder = builder.register_custom("wait", parse_wait_directive);
481    builder = builder.register_custom("end", parse_end_directive);
482    builder = builder.register_custom("routine", parse_routine_directive);
483
484    for directive in OpenAccDirective::ALL {
485        let name = directive.as_str();
486        if matches!(name, "cache" | "wait" | "end" | "routine") {
487            continue;
488        }
489        builder = builder.register_generic(name);
490    }
491
492    builder.build()
493}

499fn parse_parenthesized_content_inner(input: &str) -> nom::IResult<&str, String> {
500    use crate::lexer;
501    use nom::bytes::complete::tag;
502    use nom::error::{Error, ErrorKind};
503
504    let (input, _) = lexer::skip_space_and_comments(input)?;
505    let (input, _) = tag("(")(input)?;
506
507    let mut depth = 1;
508    let mut end_index = None;
509
510    for (idx, ch) in input.char_indices() {
511        match ch {
512            '(' => depth += 1,
513            ')' => {
514                depth -= 1;
515                if depth == 0 {
516                    end_index = Some(idx);
517                    break;
518                }
519            }
520            _ => {}
521        }
522    }
523
524    let Some(end_idx) = end_index else {
525        return Err(nom::Err::Error(Error::new(input, ErrorKind::Tag)));
526    };
527
528    let (content, rest) = input.split_at(end_idx);
529    let rest = &rest[1..];
530
531    Ok((rest, content.to_string()))
532}

534/// Parse a comma-separated list of variable names/expressions
535/// Handles: m, n, p
536///          x[0:N], y[1:10]
537///          max::x, readonly::y (C++ scope resolution)
538fn parse_variable_list(input: &str) -> Vec<Cow<'_, str>> {
539    let mut variables = Vec::new();
540    let mut current = String::new();
541    let mut depth = 0; // Track parenthesis/bracket depth for array sections
542
543    for ch in input.chars() {
544        match ch {
545            ',' if depth == 0 => {
546                let trimmed = current.trim();
547                if !trimmed.is_empty() {
548                    variables.push(Cow::Owned(trimmed.to_string()));
549                }
550                current.clear();
551            }
552            '(' | '[' => {
553                depth += 1;
554                current.push(ch);
555            }
556            ')' | ']' => {
557                depth -= 1;
558                current.push(ch);
559            }
560            _ => {
561                current.push(ch);
562            }
563        }
564    }
565
566    // Don't forget the last variable
567    let trimmed = current.trim();
568    if !trimmed.is_empty() {
569        variables.push(Cow::Owned(trimmed.to_string()));
570    }
571
572    variables
573}

575/// Parse copyin clause: copyin(readonly: m, n) or copyin(m, n)
576fn parse_copyin_clause<'a>(
577    name: Cow<'a, str>,
578    input: &'a str,
579) -> nom::IResult<&'a str, super::Clause<'a>> {
580    use crate::lexer;
581    use crate::parser::clause::{ClauseKind, CopyinModifier};
582    use nom::bytes::complete::tag;
583
584    let (input, _) = lexer::skip_space_and_comments(input)?;
585    let (input, _) = tag("(")(input)?;
586    let (input, _) = lexer::skip_space_and_comments(input)?;
587
588    // Try to parse "readonly:" modifier (case-insensitive)
589    let input_lower = input.trim_start().to_lowercase();
590    let (input, modifier) =
591        if input_lower.starts_with("readonly:") || input_lower.starts_with("readonly :") {
592            // Find the colon
593            let colon_idx = input.find(':').unwrap();
594            let after_colon = &input[colon_idx + 1..];
595            (after_colon, Some(CopyinModifier::Readonly))
596        } else {
597            (input, None)
598        };
599
600    // Parse until closing paren
601    let (input, _) = lexer::skip_space_and_comments(input)?;
602    let mut depth = 0;
603    let mut end_idx = None;
604
605    for (idx, ch) in input.char_indices() {
606        match ch {
607            '(' | '[' => depth += 1,
608            ')' if depth == 0 => {
609                end_idx = Some(idx);
610                break;
611            }
612            ')' | ']' => depth -= 1,
613            _ => {}
614        }
615    }
616
617    let end_idx = end_idx.ok_or_else(|| {
618        nom::Err::Error(nom::error::Error::new(input, nom::error::ErrorKind::Tag))
619    })?;
620
621    let var_list_str = &input[..end_idx];
622    let variables = parse_variable_list(var_list_str);
623    let rest = &input[end_idx + 1..];
624
625    Ok((
626        rest,
627        super::Clause {
628            name,
629            kind: ClauseKind::CopyinClause {
630                modifier,
631                variables,
632            },
633        },
634    ))
635}

638fn parse_copyout_clause<'a>(

759/// Parse reduction clause: reduction(+: sum) or reduction(max: x, y, z)
760fn parse_reduction_clause<'a>(
761    name: Cow<'a, str>,
762    input: &'a str,
763) -> nom::IResult<&'a str, super::Clause<'a>> {
764    use crate::lexer;
765    use crate::parser::clause::{ClauseKind, ReductionOperator};
766    use nom::bytes::complete::tag;
767
768    let (input, _) = lexer::skip_space_and_comments(input)?;
769    let (input, _) = tag("(")(input)?;
770    let (input, _) = lexer::skip_space_and_comments(input)?;
771
772    // Parse operator (everything before the first colon)
773    let colon_idx = input.find(':').ok_or_else(|| {
774        nom::Err::Error(nom::error::Error::new(input, nom::error::ErrorKind::Tag))
775    })?;
776
777    // Normalize operator to lowercase for case-insensitive matching
778    let op_str = input[..colon_idx].trim().to_lowercase();
779    let operator = match op_str.as_str() {
780        "+" => ReductionOperator::Add,
781        "-" => ReductionOperator::Sub,
782        "*" => ReductionOperator::Mul,
783        "max" => ReductionOperator::Max,
784        "min" => ReductionOperator::Min,
785        "&" => ReductionOperator::BitAnd,
786        "|" => ReductionOperator::BitOr,
787        "^" => ReductionOperator::BitXor,
788        "&&" => ReductionOperator::LogAnd,
789        "||" => ReductionOperator::LogOr,
790        ".and." => ReductionOperator::FortAnd,
791        ".or." => ReductionOperator::FortOr,
792        ".eqv." => ReductionOperator::FortEqv,
793        ".neqv." => ReductionOperator::FortNeqv,
794        "iand" => ReductionOperator::FortIand,
795        "ior" => ReductionOperator::FortIor,
796        "ieor" => ReductionOperator::FortIeor,
797        _ => {
798            return Err(nom::Err::Error(nom::error::Error::new(
799                input,
800                nom::error::ErrorKind::Tag,
801            )))
802        }
803    };
804
805    let input = &input[colon_idx + 1..];
806
807    // Check if there's a space after the colon (for formatting preservation)
808    let space_after_colon = input.starts_with(|c: char| c.is_whitespace());
809
810    // Parse until closing paren
811    let (input, _) = lexer::skip_space_and_comments(input)?;
812    let mut depth = 0;
813    let mut end_idx = None;
814
815    for (idx, ch) in input.char_indices() {
816        match ch {
817            '(' | '[' => depth += 1,
818            ')' if depth == 0 => {
819                end_idx = Some(idx);
820                break;
821            }
822            ')' | ']' => depth -= 1,
823            _ => {}
824        }
825    }
826
827    let end_idx = end_idx.ok_or_else(|| {
828        nom::Err::Error(nom::error::Error::new(input, nom::error::ErrorKind::Tag))
829    })?;
830
831    let var_list_str = &input[..end_idx];
832    let variables = parse_variable_list(var_list_str);
833    let rest = &input[end_idx + 1..];
834
835    Ok((
836        rest,
837        super::Clause {
838            name,
839            kind: ClauseKind::ReductionClause {
840                operator,
841                variables,
842                space_after_colon,
843            },
844        },
845    ))
846}
