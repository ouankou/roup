1//! Header verification binary - ensures roup_constants.h is up-to-date
2//!
3//! **Usage:** `cargo run --bin gen`
4//!
5//! This binary verifies that the committed `src/roup_constants.h` matches
6//! the constants defined in `src/c_api.rs`. It's used in CI to catch cases
7//! where the header wasn't regenerated after modifying the source.
8//!
9//! For design rationale and implementation details, see:
10//! [`docs/BUILD_SCRIPT_RATIONALE.md`](../../docs/BUILD_SCRIPT_RATIONALE.md)
11
12use std::fs;
13use std::path::Path;
14use std::process;
15
16// Use #[path] to share code with build.rs
17// This pattern is documented in docs/BUILD_SCRIPT_RATIONALE.md
18#[path = "../constants_gen.rs"]
19mod constants_gen;
20
21// Only import the functions we actually use (verification, not generation)
22use constants_gen::{
23    calculate_combined_checksum, extract_checksum_from_header, parse_acc_clause_mappings,
24    parse_acc_directive_mappings, parse_clause_mappings, parse_directive_mappings,
25};
26
27fn main() {
28    println!("Verifying src/roup_constants.h is up-to-date...");
29
30    // Step 1: Check header exists
31    if !Path::new("src/roup_constants.h").exists() {
32        eprintln!("❌ ERROR: src/roup_constants.h not found");
33        eprintln!("   Run 'cargo build' to generate the header");
34        process::exit(1);
35    }
36
37    // Step 2: Calculate expected checksum from source (using syn)
38    let directives = parse_directive_mappings();
39    let clauses = parse_clause_mappings();
40    let acc_directives = parse_acc_directive_mappings();
41    let acc_clauses = parse_acc_clause_mappings();
42    let expected_checksum =
43        calculate_combined_checksum(&directives, &clauses, &acc_directives, &acc_clauses);
44
45    println!(
46        "Calculated from source: {} (OpenMP: {} directives + {} clauses, OpenACC: {} directives + {} clauses)",
47        expected_checksum,
48        directives.len(),
49        clauses.len(),
50        acc_directives.len(),
51        acc_clauses.len()
52    );
53
54    // Step 3: Read checksum from committed header
55    let header_content =
56        fs::read_to_string("src/roup_constants.h").expect("Failed to read header file");
57
58    let header_checksum = extract_checksum_from_header(&header_content);
59
60    match header_checksum {
61        Some(checksum) => {
62            println!("Committed header:       {checksum}");
63
64            if checksum == expected_checksum {
65                println!("✓ Header is up-to-date with source code");
66                process::exit(0);
67            } else {
68                eprintln!();
69                eprintln!("❌ ERROR: Header is out of date!");
70                eprintln!("   Expected: {expected_checksum}");
71                eprintln!("   Found:    {checksum}");
72                eprintln!();
73                eprintln!("   The committed src/roup_constants.h doesn't match src/c_api.rs");
74                eprintln!("   To fix: Run 'cargo build' locally and commit the updated header");
75                process::exit(1);
76            }
77        }
78        None => {
79            eprintln!("❌ ERROR: Could not find ROUP_CONSTANTS_CHECKSUM in header");
80            process::exit(1);
81        }
82    }
83}
